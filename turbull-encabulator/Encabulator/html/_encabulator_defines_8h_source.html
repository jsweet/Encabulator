57 // LumiGeek 1X Addressable Strip MultiTool Commands
58 // There is just single header for the addressable shield on the TurBull Encabulator.
59 // Always set header = 1 in the I2C commands.
60 
61 // You can set the type of addressable strip to be driven by this shield.
62 // The default is WS2811, used in the strip that arrived with the TurBull Encabulator.
63 // StripMode: 1 = WS2811 e.g. http://adafruit.com/products/1138
64 // 2 = WS2801 e.g. http://sparkfun.com/products/11272, http://sparkfun.com/products/11020, http://adafruit.com/products/322, http://adafruit.com/products/738
65 // 3 = LPD8806 e.g. http://adafruit.com/products/306
66 
67 #define LG_ADDR_1X_CMD_SET_STRIP_MODE 0x58 // 0x58 Header StripMode
68 
69 // Here are some generative macros, limited for now to 256 pixels by the length byte
70 
71 #define LG_ADDR_1X_CMD_GRADIENT 0x51 // 0x51 Header R1 G1 B1 R2 G2 B1 Length
72 #define LG_ADDR_1X_CMD_COMET_CHASE 0x55 // 0x55 Header R G B LengthOfComet LengthOfTail Speed
73 
74 // Here is how to clock in raw RGB pixel data. Unlike the generative macros, the frame command can handle more than 256 pixels.
75 // Issues concerning power and voltage drop on long runs of strip will likely crop up before hitting any pixel limit on the I2C bus.
76 // The strip driver chips themselves may have limitations. Please see their respective datasheets. (i.e. WS2811 appears to have a 1024 pixel limit)
77 
78 #define LG_ADDR_1X_CMD_FRAME 0x53 // 0x51 Header R1 G1 B1 ... RN GN BN
79 
80 // Now that you have pixel data in our 1024 buffer on the shield, you can rotate or autorotate portions of the buffer
81 // The length of the buffer "wrap around" effect is limited to 256
82 
83 #define LG_ADDR_1X_CMD_SHIFT_BUFFER_ONCE 0x56 // 0x56 Header Direction(0=left,1=right) BufferLengthForWrap
84 #define LG_ADDR_1X_CMD_AUTOSHIFT_BUFFER 0x57 // 0x57 Header Direction(0=left,1=right) BufferLengthForWrap Speed
