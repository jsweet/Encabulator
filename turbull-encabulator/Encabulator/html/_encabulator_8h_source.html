125 // Class: XYZHelper
126 // This is a rudiemntary helper class for the KXTI9-1001 accelerometer by Kionix.
127 // It is initialized with a call to begin().
128 // Calling updateValues() will fetch the twos-compliment of the raw x, y,and z values
129 // This is a very robust accelerometer chip and direct I2C calls can access features
130 // such as tap-detection and filtering that are not exposed in this simple helper class.
131 // ---------------------------------------------------------------------------
132 
133 class XYZHelper
134 {
135  private:
136  int16_t _x;
137  int16_t _y;
138  int16_t _z;
139  bool _initialized;
140  protected:
141  void writeRegister(uint8_t reg, uint8_t data);
142  uint8_t readRegister(uint8_t reg);
143  int16_t combineBytes(uint8_t tempLow, uint8_t tempHigh);
144  public:
145  XYZHelper() {
146  _initialized = false;
147  };
148  void begin();
149  void update();
150  int16_t x();
151  int16_t y();
152  int16_t z();
153 };
